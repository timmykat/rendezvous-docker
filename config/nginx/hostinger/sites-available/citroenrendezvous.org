# ================================
# Production Citroen Rendezvous
# ================================

# --------------------------------
# HTTP: redirect all traffic to HTTPS
# --------------------------------

# Main domain redirect
server {
    listen 80;
    listen [::]:80;
    server_name citroenrendezvous.org www.citroenrendezvous.org;

    # Redirect all requests to HTTPS main domain
    return 301 https://citroenrendezvous.org$request_uri;
}

# Other domains redirect to main domain
server {
    listen 80;
    listen [::]:80;
    server_name citroenrendezvous.com citroenrendezvous.net citroen-rendezvous.org;

    return 301 https://citroenrendezvous.org$request_uri;
}

# Optional: catch-all wildcard domains (if needed)
server {
    listen 80 default_server;
    listen [::]:80 default_server;
    server_name _;
    return 301 https://citroenrendezvous.org$request_uri;
}

# --------------------------------
# HTTPS: main site serving Rails
# --------------------------------

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;

    server_name citroenrendezvous.org www.citroenrendezvous.org;

    root /var/www/html;

    access_log /var/log/nginx/prod/access.log;
    error_log /var/log/nginx/prod/error.log info;

    # SSL certificates (replace paths if needed)
    ssl_certificate /etc/letsencrypt/live/citroenrendezvous.org-0001/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/citroenrendezvous.org-0001/privkey.pem;

    ssl_session_cache shared:SSL:1m;
    ssl_session_timeout 5m;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305:DHE-RSA-AES128-GCM-SHA256:DHE-RSA-AES256-GCM-SHA384:DHE-RSA-CHACHA20-POLY1305;

    # Rails proxy
    location / {
        proxy_pass http://rails_app_prod;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Host $http_host;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_redirect off;

        add_header Cache-Control "no-cache, no-store, must-revalidate" always;
        add_header Pragma "no-cache" always;
        add_header Expires 0 always;
    }

    # Assets
    location ~ ^/(assets|packs|files|uploads)/ {
        gzip_static on;
        expires max;
        add_header Cache-Control "public, max-age=31536000, immutable";
        try_files $uri =404;
        error_page 404 /404.html;
    }

    # Other static files
    location ~ ^(?!/rails/).+\.(jpg|jpeg|gif|png|ico|json|txt|xml|csv|pdf)$ {
        gzip_static on;
        expires max;
        add_header Cache-Control "public, max-age=31536000, immutable";
        try_files $uri =404;
        error_page 404 /404.html;
    }

    # Maintenance mode
    if (-f $document_root/files/show_maintenance.txt) {
        return 503;
        break;
    }

    error_page 503 @maintenance;
    location @maintenance {
        rewrite ^(.*)$ /files/maintenance.html break;
    }

    # Fix double slashes in URLs
    merge_slashes off;
    rewrite ^(.*?)//+(.*?)$ $1/$2 permanent;
}

# --------------------------------
# HTTPS: redirect other domains to main domain
# --------------------------------

server {
    listen 443 ssl;
    listen [::]:443 ssl;

    server_name citroenrendezvous.com citroenrendezvous.net citroen-rendezvous.org;

    ssl_certificate /etc/nginx/ssl/certs/prod/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/certs/prod/privkey.pem;

    return 301 https://citroenrendezvous.org$request_uri;
}

# Optional: catch-all wildcard redirects
server {
    listen 443 ssl default_server;
    listen [::]:443 ssl default_server;

    server_name _;

    ssl_certificate /etc/nginx/ssl/certs/prod/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/certs/prod/privkey.pem;

    return 301 https://citroenrendezvous.org$request_uri;
}
